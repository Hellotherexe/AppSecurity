/* Quick Reference: Adding Password Validation to Any Form */

/* 
 * STEP 1: Add to your view/page
 * Simply add the data-password-validation attribute to your password input
 */

<input type="password" 
       class="form-control" 
       id="Password" 
       name="Password" 
       data-password-validation
       required />

/*
 * STEP 2: Include the JavaScript in your view
 * Add this in the Scripts section at the bottom of your view
 */

@section Scripts {
    <script src="~/js/password-validation.js" asp-append-version="true"></script>
}

/*
 * STEP 3: Ensure the CSS is loaded
 * The CSS is already included in _Layout.cshtml
 * If not using the layout, add this to your view's head section:
 */

<link rel="stylesheet" href="~/css/password-validation.css" asp-append-version="true" />

/*
 * STEP 4: Server-side validation in your controller
 * ALWAYS validate on the server - never trust client-side validation alone!
 */

[HttpPost]
public IActionResult YourAction(string password)
{
    // Inject PasswordPolicyService in your constructor
    var validation = _passwordService.ValidatePassword(password);
    
    if (!validation.IsValid)
    {
        foreach (var error in validation.ErrorMessages)
        {
            ModelState.AddModelError("Password", error);
        }
        return View();
    }
    
    // Password is valid, continue processing
}
